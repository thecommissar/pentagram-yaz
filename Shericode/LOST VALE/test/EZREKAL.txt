process Ezrekal::schedule(currentTime:Long)
{

	if inShadeScene = true
	{
		return;
	}

	point : WorldPoint;
	map : MapNum = 10;


	//inside

	if (getPeriod() = daytide) or (getPeriod() = eventide)
	or (getPeriod() = bloodwatch) or (getPeriod() = threemoons)
	{
		point.x = 19327;
		point.y = 29439;
		point.z = 64;
	}



	// outside

	if (getPeriod() = firstebb) or (getPeriod() = lastebb)
	{
		point.x = 18431;
		point.y = 29311;
		point.z = 64;
	}



	safeTeleport(point,map);
}

process Ezrekal::enterFastArea()
{

  	if isDead()
	{
		return;
	}

	if inShadeScene = true
	{
		killProcess(referent, PT_ACTIVITY);
		return;
	}

	cSetActivity(ACT_LOITER);
}


process Ezrekal::leaveFastArea()
{

	justGotRoot = false;
	resetRef(referent,PT_ACTIVITY);
	resetRef(referent,PT_PATHFIND);
	resetRef(referent,PT_ANIM);
	resetRef(referent,PT_BARK);
}


process Ezrekal::cast(ref:Referent)
{
	count:Number;
	count = 10;
	while (not getTarget()) and (count)
	{
		wait(1);
		count--;
	}

	ref=getTarget();
	if ref = Avatar:Referent
	{
		spawn callTheGuards(NULL_REF) -> return;

		if not isDead()
		{
			killProcess(referent,PT_ACTIVITY);
			killProcess(referent,PT_BARK);
			bark("Help! Help! \g Hilfe! Hilfe!");
		}
	}
}
