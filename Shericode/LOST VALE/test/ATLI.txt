process Atli::schedule(currentTime:Long)
{
	if inShadeScene = true
	{
		return;
	}

	if plantScene = true
	{
		return;
	}

	point : WorldPoint;
	map : MapNum = 10;


	//Eating

	if (getPeriod() = daytide) or (getPeriod() = eventide)
	or (getPeriod()= threemoons)
	{
		point.x = 18943;
		point.y = 19711;
		point.z =64;
	}



	//Lab 1

	if (getPeriod() = firstebb) or (getPeriod() = lastebb)
	or (getPeriod()=bloodwatch)
	{
		point.x = 18175;
		point.y = 20095;
		point.z = 64;
	}


	safeTeleport(point,map);
}

process Atli::enterFastArea()
{
	if isDead()
	{
		return;
	}

	if inShadeScene = true
	{
		killProcess(referent, PT_ACTIVITY);
		return;
	}

	cSetActivity(ACT_LOITER);
}


process Atli::leaveFastArea()
{
	killProcess(referent,PT_ACTIVITY);
}


process Atli::cast(ref:Referent)
{
	count:Number;
	count = 10;
	while (not getTarget()) and (count)
	{
		wait(1);
		count--;
	}

	ref=getTarget();
	if ref = Avatar:Referent
	{
		spawn callTheShades() -> return;

		if not isDead()
		{
			killProcess(referent,PT_ACTIVITY);
			killProcess(referent,PT_BARK);
			bark("Help! Help! \f A l'aide! A l'aide! \g");
		}
	}
}
